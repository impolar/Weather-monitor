Database	weatherLog

Juoksevat mittaustulokset

CREATE TABLE dataLog (
id INT PRIMARY KEY AUTO_INCREMENT,
date DATETIME,
temperature FLOAT, 
humidity FLOAT ) Engine=InnoDB;

Vuorokauden maksimi- ja minimiarvot

CREATE TABLE historyLog (
id INT PRIMARY KEY AUTO_INCREMENT,
date DATETIME,				
temperature_min FLOAT,
temperature_max FLOAT,
humidity_min FLOAT,
humidity_max FLOAT ) Engine=InnoDB; 

// Haku snipeti snäppejä varten
SELECT date AS 'Päivämäärä', temperature_min AS 'Alin lämpötila', temperature_max AS 'Ylin lämpötila', humidity_min AS 'Alin kosteusprosentti', humidity_max AS 'Ylin kosteusprosentti' FROM historyLog;

SELECT date AS 'Päivämäärä', temperature AS 'Lämpötila, C°', humidity AS 'Kosteusprosentti, %' FROM dataLog;
